BibTeX Entry

To reference an individual chapter:

@InCollection{AuthorLastName15,
  author = {Author(s) Name},
  title = {Chapter Title},
  booktitle = {{W}eb{GL} {I}nsights},
  pages = {First Page-Last Page},
  editor = {Patrick Cozzi},
  month = {August},
  year = {2015},
  isbn = {978-1498716079},
  publisher = {CRC Press},
  note = {\url{http://www.webglinsights.com/}}
}

To reference the entire book:

@book{Cozzi15,
  author = {Patrick Cozzi},
  title = {{W}eb{GL} {I}nsights},
  month = {August},
  year = {2015},
  isbn = {978-1498716079},
  publisher = {CRC Press}
  note = {\url{http://www.webglinsights.com/}}
}
